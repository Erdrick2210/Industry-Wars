# modules.txt
# Formato:
# [ID]
# Name=...
# Type=passive|active
# Category=offensive|defensive|energetic|tactical
# Icon=...
# Modifiers=stat,val,stat,val,...   (vacío si ninguno)
# PassiveID=ID_EFECTO_PASIVO        (solo pasivos; resuelto en combate)
# Skill=ID_HABILIDAD                (solo activos; definida en abilities.txt)
# SkillCost=N                       (EP; sobreescribe el coste base de la habilidad)
# SkillCooldown=N                   (turnos; sobreescribe el cooldown base)
# RequiredCore=ID_NUCLEO            (vacío = sin restricción)
# StatSlot=attack|defense|speed|none
